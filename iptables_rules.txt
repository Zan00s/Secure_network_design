#!/bin/bash

iptables -F FORWARD

#SMTP

iptables -A FORWARD -s 10.10.10.0/24 -p tcp --dport 25 \
        -m time --timestart 07:00 --timestop 22:59 --weekdays Mon \
        -j ACCEPT
iptables -A FORWARD -s 10.10.10.0/24 -p tcp --dport 25 -j DROP


#RDP

iptables -A FORWARD -s 10.10.10.0/24 -p tcp --dport 3389 -j DROP

#SSH

iptables -A FORWARD -s 10.10.10.0/24 -p tcp --dport 22 -j DROP

#save
iptables-save > /etc/iptables/rules.v4

echo "Firewall rules applied successfully."
